\section{Probability Distributions}
\prob{
	Prove that
	\begin{align}
		\left(\begin{matrix}\A & \B \\ \C & \D\end{matrix}\right)^{-1} = \left(\begin{matrix}\M & -\M\B\D^{-1} \\ -\D^{-1}\C\M & \D^{-1}+\D^{-1}\C\M\B\D^{-1} \end{matrix}\right)
	\end{align}
	where $\M = (\A - \B\D^{-1}\C)^{-1}$.
}
\sol{
	We apply Gaussian Elimianation in the following way:
	\begin{align}
		\left[ \left(\begin{matrix}\I & \O \\ -\D^{-1}\C & \I\end{matrix}\right) \left(\begin{matrix}(\A-\B\D^{-1}C)^{-1} & \O \\ \O & \I\end{matrix}\right) \left(\begin{matrix}\I & -\B \\ \O & \I\end{matrix}\right) \left(\begin{matrix}\I & \O \\ \O & \D^{-1}\end{matrix}\right) \right] \left(\begin{matrix}\A & \B \\ \C & \D\end{matrix}\right) = \left(\begin{matrix}\I & \O \\ \O & \I\end{matrix}\right) 
	\end{align}
	and we have
	\begin{align}
		\left(\begin{matrix}\A & \B \\ \C & \D\end{matrix}\right)^{-1} &= \left(\begin{matrix}\I & \O \\ -\D^{-1}\C & \I\end{matrix}\right) \left(\begin{matrix}(\A-\B\D^{-1}C)^{-1} & \O \\ \O & \I\end{matrix}\right) \left(\begin{matrix}\I & -\B \\ \O & \I\end{matrix}\right) \left(\begin{matrix}\I & \O \\ \O & \D^{-1}\end{matrix}\right) \\
		&= \left(\begin{matrix}(\A-\B\D^{-1}\C)^{-1} & -(\A-\B\D^{-1}\C)^{-1}\B\D^{-1} \\ -\D^{-1}\C(\A-\B\D^{-1}\C)^{-1} & \D^{-1}+\D^{-1}\C(\A-\B\D^{-1}\C)^{-1}\B\D^{-1} \end{matrix}\right)
	\end{align}
}

\prob{
	Show that the inverse of a symmetric matrix is itself symmetric.
}
\sol{
	Suppose $\A = \A^T$ is a symmetric matrix, and invertible. We have
	\begin{align}
		(\A^{-1})^T = (\A^{-1})^T\A\A^{-1} = (\A^{-1})^T\A^T\A^{-1} = (\A\A^{-1})^T \A^{-1} = \A^{-1}
	\end{align}
}
