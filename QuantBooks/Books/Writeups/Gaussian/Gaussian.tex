\section{Univariate Gaussian Distribution}
\begin{align}
	\phi'(x) &= -x\phi(x)\\
	\int \Phi(x) dx &= x\Phi(x)+\phi(x)
\end{align}

\section{Multivariate Gaussian Distribution}
\begin{align}
	\phi'(x) &= -x\phi(x)\\
	\int \Phi(x) dx &= x\Phi(x)+\phi(x)
\end{align}
